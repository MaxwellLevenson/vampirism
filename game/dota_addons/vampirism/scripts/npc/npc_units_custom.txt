  //=================================================================================//
  // Vampirism Naming Convention - Please read before making changes
  //
  //   Custom units should follow the following convention.
  //   class_type_tier 
  //   - type is optional if there is only one in a class, as in slayer_tavern.
  //   - tier is optional if there is only one in a tier, as in tower_frost
  //
  //   Examples of good naming:
  //
  //   tower_pearls_t1
  //   - unit class is tower, type is pearls and tier is t1
  //
  //   house_t1
  //   - unit class is house, type is not needed and tier is t1.
  //
  //   Examples of bad naming:
  //
  //  shiny_tower_pearls
  //   - named according to its unit name, not really the best way to do this.       
  // 
  //   To indicate a units real name / the name that is shown to the player, place   
  //   it in the ascii header above the unit.
  //
  // Custom Keys
  //    
  //    These are custom keys that are used by various components of the game.
  //    
  //     "AffectedByGemUpgrades" "true" -- Whether a unit will be affected by the basic gem upgrades.
  //     "RepairTime"            "15" -- The amount of time to repair this from 0-100% HP.
  //     "IsTech"                "true" -- Whether this unit should be tracked by TechTree.
  //     "NeedTech"              "house_t1 wall_t1" -- Any buildings that are needed to build this unit, seperated by spaces.
  //     "SpawnsUnits"           "true"  -- Whether this unit can spawn harvesters/workers.
  //     "TechModifiers"
  //     {
  //        "1"             "research_healing_vitality" -- A list of research upgrades that this unit is affected by.
  //     }
  //     "ParentUnit"       "house_t2" -- The unit that this unit was made from.
  //
  //=================================================================================//
